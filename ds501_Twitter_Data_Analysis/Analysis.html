Please briefly describe in the following cell: what problem are you trying to solve? why this problem is important and interesting?

Tesla is an all-electric car manufacturer and Model 3 is their first product in mid-size sedan at an affordable price. The demand has been huge since day 1 of the car was unveiled. However, due to the limit of battery technology, the production has been far behind the demand, and many related issues came out. For example, quality flaws, and high turnover rate in top executives. Since beginning of 2018, the production of Model 3 has significant improvement and some buyers gradually picked up their new cars. To further improve the strategies for car design and sales, we want to explore the market feedback for Model3: How do people like Model3? What do owners and potential buyers keep an eye on about Model3?

def createSentimentalDF(pos_neg, subj_obj, tweets, tweets_cleaned, create_time):
    pos_neg_class = []
    for v in pos_neg:
        if v > 0.1:
            pos_neg_class.append('Positive')
        elif v < -0.1:
            pos_neg_class.append('Negative')
        else:
            pos_neg_class.append('Neutral')

    subj_obj_class = []
    for v in subj_obj:
        if v > 0.5:
            subj_obj_class.append('Subjective')
        elif v <= 0.5:
            subj_obj_class.append('Objective')


    sentimental_df = pd.DataFrame(data = {
        'Create_Time': create_time,
        'Full_Text': tweets,
        'Full_Text Cleaned': tweets_cleaned,
        'Polarity':pos_neg,
        'Polarity_Class': pos_neg_class,
        'Subjectivity':subj_obj,
        'Subjectivity_Class': subj_obj_class,
    })
    
    return sentimental_df
def splitIntoPosNeg(sentimental_df):
    # split sentimental_df
    # pos_df, pos_text, neg_df, neg_text
    pos_df = sentimental_df[
        sentimental_df.Polarity_Class == 'Positive'
    ]
    neg_df = sentimental_df[
        sentimental_df.Polarity_Class == 'Negative'
    ]

    pos_text = ''
    neg_text = ''
    for txt in neg_df.Full_Text:
        neg_text += txt
    for txt in pos_df.Full_Text:
        pos_text += txt
        
    return pos_df, pos_text, neg_df, neg_text


from nltk.corpus import stopwords

def cleanTextData(text):
    #Input: text
    #Output: cleaned word list
    
    # lower words
    text = text.lower()
    
    # remove urls
    text = re.sub(r'http\S+', '', text)
    
    # remove punctuation and tokenize
    tokenizer = RegexpTokenizer(r'\w+')
    wordTokenized = tokenizer.tokenize(text)
    
    # remove stopwords
    stop_words = set(stopwords.words('english')) 
    filtered_sentence = [] 
    for w in wordTokenized: 
        if w not in stop_words and w != 'rt' and w != 'RT' and not w.startswith('@')\
         and not w.startswith('#') and w != '3' and w!= 'tesla' and w!='elonmusk' and w!='model3'\
         and w!='teslamodel3' and w!='amp' and w!='model' : 
            filtered_sentence.append(w) 
    return filtered_sentence


def wordCountAndSort(wordList):
    #Input wordlist
    #Output dataframe with sorted word count
    wordCnt = dict(Counter(wordList))
    df = pd.DataFrame(sorted(wordCnt.items(), key=itemgetter(1), reverse = True))
    df.columns = ['word','count']
    
    return df

def wordcloud_draw(data, color = 'white'):
    # input: cleeaned text (string)
    words = ' '.join(data)
    cleaned_word = " ".join([word for word in words.split()
                            if 'http' not in word
                                and not word.startswith('@')
                                and not word.startswith('#')
                                and word != 'RT' and word != 'rt'
                            ])
    wordcloud = WordCloud(stopwords=STOPWORDS,
                      background_color=color,
                      width=2500,
                      height=2000
                     ).generate(cleaned_word)
    plt.figure(1,figsize=(13, 13))
    plt.imshow(wordcloud)
    plt.axis('off')
    plt.show()
