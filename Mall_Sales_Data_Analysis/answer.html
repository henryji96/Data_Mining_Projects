def parseOutDate(dt):
    if dt is np.nan:
        return dt           #如果日期是nan，则返回nan
    dt = dt.split(' ')[0]
    year = int(dt[0:4])
    if year > 2020:         #有一行年>2020，为异常值，设为nan
        return np.nan
    return datetime.strptime(dt, '%Y/%m/%d')
